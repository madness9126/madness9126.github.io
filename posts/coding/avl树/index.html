Avl树

AVL树本质上还是一棵二叉搜索树，它的特点是：

  1. 本身首先是一棵二叉搜索树。
  2. 带有平衡条件：每个结点的左右子树的高度之差的绝对值（平衡因子）最多为1。 也就是说，AVL树，本质上是带了平衡功能的二叉查找树（二叉排序树，二叉搜索树）。
#include <stdio.h>
#include <stdlib.h>

typedef struct avl
{
    struct avl *left;
    struct avl *right;
    int key;
    int high;
} avl, *avlp;

int max(int a, int b)
{
    return a > b ? a : b;
}

void avllist(avlp root)
{
    if (root)
    {
        printf("key:%d high:%d\n", root->key, root->high);
        avllist(root->left);
        avllist(root->right);
    }
}

int avlhigh(avlp tree)
{
    if (!tree)
    {
        return -1;
    }
    else
        return max(avlhigh(tree->left), avlhigh(tree->right)) + 1;
}

// ll
avlp right_rotation(avlp tree)
{
    avlp node = tree->left;
    tree->left = node->right; //保存废弃节点
    node->right = tree;       //连接

    //计算树高
    tree->high = avlhigh(tree);
    node->high = avlhigh(node);
    return node;
}

// RR
avlp left_rotation(avlp tree)
{
    avlp node = tree->right;
    tree->right = node->left;
    node->left = tree;

    tree->high = avlhigh(tree);
    node->high = avlhigh(node);
    return node;
}

// lR
avlp left_right_rotation(avlp tree)
{
    tree->left = left_rotation(tree->left);
    return right_rotation(tree);
}

// Rl
avlp right_left_rotation(avlp tree)
{
    tree->right = right_rotation(tree->right);
    return left_rotation(tree);
}

avlp rebalance(avlp tree, int key)
{
    int factor = avlhigh(tree->left) - avlhigh(tree->right);
    if (factor > 1)
    {
        if (key > tree->left->key)
        {
            printf("left_right_rotation \n");
            tree = left_right_rotation(tree);
        }
        else
        {
            printf("right_rotation \n");
            tree = right_rotation(tree);
        }
    }
    if (factor < -1)
    {
        if (key > tree->right->key)
        {
            printf("left_rotation \n");
            tree = left_rotation(tree);
        }
        else
        {
            printf("right_left_rotation \n");
            tree = right_left_rotation(tree);
        }
    }
    return tree;
}

avlp avlinsert(avlp tree, int value)
{
    if (!tree)
    {
        avlp node = (avlp)malloc(sizeof(avl));
        if (!node)
        {
            printf("node is malloc error");
            return NULL;
        }
        node->key = value;
        node->high = 0;

        node->left = NULL;
        node->right = NULL;
        return node;
    }
    if (value >= tree->key)
    {
        tree->right = avlinsert(tree->right, value); //从rigth递去,tree->表示连接
        tree->high = avlhigh(tree);
    }
    else
    {
        tree->left = avlinsert(tree->left, value); //从left递去
        tree->high = avlhigh(tree);
    }
    tree = rebalance(tree, value);
    return tree;
}

avlp findmin(avlp tree)
{
    if (!tree)
    {
        return NULL;
    }
    if (!tree->left)
    {
        return tree;
    }
    else
        return findmin(tree->left);
}

avlp findmax(avlp tree)
{
    if (!tree)
    {
        return NULL;
    }
    if (!tree->right)
    {
        return tree;
    }
    else
        return findmax(tree->right);
}

avlp avldelete(avlp tree, int key)
{
    if (!tree)
    {
        return NULL;
    }
    if (key > tree->key)
    {
        tree->right = avldelete(tree->right, key);
        tree = rebalance(tree,key);
    }
    if (key < tree->key)
    {
        tree->left = avldelete(tree->left, key);
        tree = rebalance(tree,key);
    }
    if (key == tree->key)
    {
        if (tree->left && tree->right)
        {
            if (avlhigh(tree->left) > avlhigh(tree->right))
            {
                avlp max = findmax(tree->left); //找出左子树最大值
                tree->key = max->key;           // swap
                tree->left = avldelete(tree->left, max->key);
            }
            else
            {
                avlp min = findmin(tree->right);
                tree->key = min->key;
                tree->right = avldelete(tree->right, min->key);
            }
        }
        else{
            avlp node = NULL;
            node = tree;
            if (!tree->left)
            {
                tree = tree->right;
            }
            else if(!tree->right) tree = tree->left;
            else tree = NULL;
        }
    }
    return tree;
}

    avlp tree = NULL;

    tree = avlinsert(tree,10);
    tree = avlinsert(tree,12);
    tree = avlinsert(tree,14);
    tree = avlinsert(tree,6);
    tree = avlinsert(tree,5);
    tree = avlinsert(tree,1);
    printf("***前序***:  \n");
    avllist(tree);
    tree = avldelete(tree,10);
    printf("***前序***:  \n");
    avllist(tree);
}
