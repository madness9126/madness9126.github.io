#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <unistd.h>

typedef struct Date
{
	int numb;
	char *name;
}Date;

typedef struct Node
{
	int date;
	struct Node *point;
}Node;

int LenthList(Node *head)
{
	int i;
	Node *temp = head;
	while(temp)
	{
		temp = temp->point;
		i++;
	}
	return i;
}

void PrintList(Node *head)
{
	Node *temp = head;

	while(temp)
	{
		printf("%d ",temp->date);
		temp = temp->point;
	}
	printf("\n");
}

Node *CraeteList(int n)
{
	Node *head = (Node *)malloc(sizeof(Node));
	head->point = NULL;
	head->date = 0;
	Node *temp = head;
	for(int i = 1;i < n;i++)
	{
		Node *node = (Node *)malloc(sizeof(Node));
		node->point = NULL;
		node->date = i;
		
		temp->point = node;
		temp = node;
	}
	return head;
}

Node *InsertList(Node *head , int n ,int numb)
{
	if(n > LenthList(head)+1 || n < 1)
	{
		return head;
	}
	Node *temp = (Node *)malloc(sizeof(Node));
	temp->point = NULL;
	temp->date = numb;
	if(n == 1)
	{
		Node *nhead = (Node *)malloc(sizeof(Node));
		nhead->point = head;
		temp->point = head;
		nhead->point = temp;
		head = nhead;
		free(nhead);
		return head;
	}
	if(n > 1 && n <= LenthList(head))
	{
		Node *temp1 = head;
		while (temp1->point != NULL && n > 2)
		{
			temp1 = temp1->point;
			n--;
		}
		temp->point = temp1->point;
		temp1->point = temp;
		return head;
	}
	if (n == LenthList(head)+1)
	{
		Node *temp2 = head;
		while (temp2!=NULL)
		{
			if (temp2->point = NULL)
			{
				break;
			}
			temp2 = temp2->point;
		}
		temp = temp2->point;
		return head;
	}
	return head;
}

Node *CopyList(Node *head)
{
	Node *nhead = (Node *)malloc(sizeof(Node));
	nhead->point = NULL;
	nhead->date = head->date;

	Node *temp = head;
	Node *ntemp = nhead;
	while (temp->point != NULL)
	{
		Node *node = (Node *)malloc(sizeof(Node));
		node->point = NULL;

		ntemp->point = node;
		ntemp = node;
		temp = temp->point;
		ntemp->date = temp->date;
	}
	return nhead;
}

Node *DeleteList(Node *head , int n)
{
	if(n <= 0 || n > LenthList(head))
	{
		return head;
	}
	if(n == 1)
	{
		Node *temp = head;
		temp = temp->point;
		head = temp;
		return head;
	}
	if(n == LenthList(head))
	{
		Node *temp = head;
		for(int i = 1 ; i < n+1 ; i++)
		{
			temp = temp->point;
		}
		temp->point = NULL;
		return head;
	}
}

int main()
{
	Node *head = CraeteList(5);
	PrintList(head);

	Node *head2 = CopyList(head);
	PrintList(head2);

	printf("LenthList = %d",LenthList(head));
	
	Node *head3 = DeleteList(head,5);
	PrintList(head3);

	Node *head1 = InsertList(head,5,10);
	PrintList(head1);
	return 0;
}
